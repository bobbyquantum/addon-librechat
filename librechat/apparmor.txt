#include <tunables/global>

profile librechat flags=(attach_disconnected,mediate_deleted,complain) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # Network capabilities
  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,
  network raw,

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json and other files within your addon
  /data/** rw,
  /share/** rw,
  /addon_config/** rw,

  # NodeJS & NPM
  /usr/local/bin/node ix,
  /usr/local/lib/node_modules/** rm,
  /opt/** rwix,

  # MongoDB
  /usr/bin/mongod cx -> mongodb,
  /var/lib/mongodb/** rwk,
  /var/log/mongodb/** rwk,

  # MeiliSearch
  /opt/meilisearch cmx -> meilisearch,
  /var/lib/meilisearch/** rwk,

  profile mongodb flags=(attach_disconnected,mediate_deleted,complain) {
    #include <abstractions/base>
    
    signal (receive) peer=*_librechat,
    
    /usr/bin/mongod r,
    /var/lib/mongodb/** rwk,
    /var/log/mongodb/** rwk,
    /data/db rwk,
    /data/db/ rwk,
    /data/db/** rwk,
    /data/db/journal/** rwk,
    /etc/gnutls/config r,
    /etc/ssl/openssl.cnf r,
    /sys/fs/cgroup/memory/memory.limit_in_bytes r,
    /sys/block/ r,
    /sys/block/** r,
    /proc/** r,
    /tmp/mongodb-*.sock rw,

    /sys/kernel/mm/transparent_hugepage/enabled r,
    /sys/kernel/mm/transparent_hugepage/defrag r,
    /run/s6/legacy-services/mongodb/** rwk,
    /etc/lsb-release r,
    network tcp,
  }

  profile meilisearch flags=(attach_disconnected,mediate_deleted,complain) {
    #include <abstractions/base>
    
    signal (receive) peer=*_librechat,
    
    /opt/meilisearch rm,
    /var/lib/meilisearch/** rwk,
    /data/meilisearch rwk,
    /data/meilisearch/ rwk,
    /data/meilisearch/** rwk,
    /root/ rw,
    /root/.config/Meilisearch/ rw,
    /root/.config/Meilisearch/** rwk,
    /run/s6/legacy-services/meilisearch/dumps/ rw,
    
    /run/s6/legacy-services/meilisearch/dumps/
    /proc/** r,
    /sys/fs/** r,
    /etc/nsswitch.conf r,
    /etc/host.conf r,
    /etc/resolv.conf r,
    /etc/hosts r,
    /etc/gai.conf r,

    network inet stream,
    network inet6 stream,
    network inet dgram,
    network inet6 dgram,
    network netlink raw,
    network tcp,
  }
}
