#include <tunables/global>

profile librechat flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # Network capabilities
  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json and other files within your addon
  /data/** rw,
  /share/** rw,
  /addon_config/** rw,

  # NodeJS & NPM
  /usr/local/bin/node ix,
  /usr/local/lib/node_modules/** rm,
  /opt/** rwix,

  # MongoDB
  /usr/bin/mongod cx -> mongodb,
  /var/lib/mongodb/** rwk,
  /var/log/mongodb/** rwk,

  # MeiliSearch
  /usr/bin/meilisearch cx -> meilisearch,
  /var/lib/meilisearch/** rwk,

  profile mongodb flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>
    
    signal (receive) peer=*_librechat,
    
    /usr/bin/mongod r,
    /var/lib/mongodb/** rwk,
    /var/log/mongodb/** rwk,

    /etc/gnutls/config r,
    /etc/ssl/openssl.cnf r,
    /sys/fs/cgroup/memory/memory.limit_in_bytes r,
    /proc/version r,
    /proc/*/mountinfo r,
    /proc/sys/net/ipv4/tcp_fastopen r,
    /tmp/mongodb-*.sock rw,

    /sys/kernel/mm/transparent_hugepage/enabled r,
    /sys/kernel/mm/transparent_hugepage/defrag r,
    /run/s6/legacy-services/mongodb/** rwk,
    /etc/lsb-release r,
    network tcp,
  }

  profile meilisearch flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>
    
    signal (receive) peer=*_librechat,
    
    /usr/bin/meilisearch r,
    /var/lib/meilisearch/** rwk,
    network inet stream,
    network inet6 stream,
    network inet dgram,
    network inet6 dgram,
    network netlink raw,
    network tcp,
  }
}
