#include <tunables/global>

profile librechat flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/openssl> 

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),

  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,
  network netlink raw,

  /etc/ssl/openssl.cnf r,
  /etc/ssl/certs/** r,
  /usr/lib/ssl/openssl.cnf r,

  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  /data/** rw,
  /share/** rw,
  /addon_config/** rw,

  /usr/local/bin/node ix,
  /usr/local/lib/node_modules/** rm,
  /opt/** rwix,

  # MongoDB
  /usr/bin/mongod cx -> mongodb,
  /var/lib/mongodb/** rwk,
  /var/log/mongodb/** rwk,

  # MeiliSearch
  /usr/bin/meilisearch cx -> meilisearch,
  /var/lib/meilisearch/** rwk,

  profile mongodb flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>
    
    signal (receive) peer=*_librechat,
    
    # MongoDB binary and data
    /usr/bin/mongod mr,
    /var/lib/mongodb/** rwk,
    /data/db/** rwkl,
    /data/db/mongod.lock rwcl,
    
    # System files
    /sys/kernel/mm/transparent_hugepage/enabled r,
    /sys/kernel/mm/transparent_hugepage/defrag r,
    /etc/lsb-release r,
    /etc/os-release r,
    /proc/*/mounts r,
    /sys/devices/system/node/ r,
    /sys/devices/system/node/node*/meminfo r,
    
    # Add missing permissions from error logs
    /etc/ssl/openssl.cnf r,
    /sys/fs/cgroup/memory/memory.limit_in_bytes r,
    /proc/version r,
    /proc/*/mountinfo r,
    /proc/sys/net/ipv4/tcp_fastopen r,
    /etc/gnutls/config r,
    
    # Socket permissions
    /tmp/mongodb-27017.sock rwcl,
    /run/s6/legacy-services/mongodb/core rwcl,
    
    # Required capabilities
    capability mknod,
    capability net_bind_service,
    
    # Network access
    network inet stream,
    network inet6 stream
  }

  profile meilisearch flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>
    
    signal (receive) peer=*_librechat,
    
    /usr/bin/meilisearch r,
    /var/lib/meilisearch/** rwk,
    network tcp,
  }

}
